package com.nit.repository;

import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.util.ArrayList;
import java.util.List;

import javax.sql.DataSource;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Repository;

import com.nit.entity.Product;
@Repository
public class ProductDAO {

	@Autowired
	DataSource dataSource;
//	insertProduct(Product product): Method to insert a new product record into the database.
//	List<Product> getAllProducts(): Method to retrieve all product records from the database.
//	deleteProduct(int productId): Method to delete a product by its ID.
//	updateProduct(Product product): Method to update an existing product.

	public String insertProduct(Product product)
	{
		String str=null;
		try {
			Connection con = dataSource.getConnection();
		    PreparedStatement ps=con.prepareStatement("insert into product51 values(?,?,?,?)");
		    ps.setInt(1, product.getProductId());
		    ps.setString(2, product.getProductName());
		    ps.setString(3, product.getDescription());
		    ps.setDouble(4, product.getPrice());
		    int executeUpdate = ps.executeUpdate();
		    if(executeUpdate>0)
		    {
		    	str="Product Inserted SuccessFully With Product Id:"+product.getProductId();
		    }
		}
		catch (Exception e) {
			e.printStackTrace();
		}
		return str;
	}
	public List<Product> getAllProducts()
	{
		List<Product> list=new ArrayList<Product>();
		try {
			Connection con = dataSource.getConnection();
		    PreparedStatement ps=con.prepareStatement("select * from product51");
		    ResultSet rs = ps.executeQuery();
		    while(rs.next())
		    {
		    	list.add(new Product(rs.getInt(1), rs.getString(2), rs.getString(3), rs.getDouble(4)));
		    }
		    
		}
		catch (Exception e) {
			e.printStackTrace();
		}
		return list;
	}
}
